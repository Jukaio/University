#version 330

layout(location = 0) in vec3 position;
layout(location = 1) in vec2 texcoord;
layout(location = 2) in vec3 normal;

uniform mat4 projection;
uniform mat4 world;
uniform mat4 view;


out vec2 f_texcoord;

out vec3 f_normal;
out vec4 f_frag_pos;

void main()
{
    f_frag_pos =  world * vec4(position, 1.0f);
    gl_Position = projection * view * world * vec4(position, 1);

    f_texcoord = texcoord;
    vec4 norm = transpose(inverse(world)) * vec4(normal, 1.0);
    f_normal = norm.xyz;
}
